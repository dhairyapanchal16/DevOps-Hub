üêö Bash Scripting Tips

#!/bin/bash
# Error handling
set -e                    # Exit on error
set -u                    # Exit on undefined variable

# Variables
VAR="value"
echo "${VAR:-default}"    # Use default if empty

# Functions
function deploy() {
    local env=$1
    echo "Deploying to $env"
}

# Loops
for file in *.log; do
    echo "Processing $file"
done
